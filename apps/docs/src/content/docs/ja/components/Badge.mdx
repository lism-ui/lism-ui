## Usage

export const ColorDemos = ({variant = '',  isColored}) => {
	const badgeProps = { variant };

	return (
		<Flex variant='cluster' g='20'>
			{['red', 'orange', 'yellow', 'green', 'blue', 'purple', 'pink', 'gray'].map((_col) => {
				if(isColored) {
					badgeProps.isColored = true; badgeProps.keycolor = _col;
				} else if(variant === 'outline') {
					badgeProps.c = _col;
				} else {
					badgeProps.bgc = _col;
				}

				return (
					<Badge key={_col}  {...badgeProps}>{_col}</Badge>
				);
			})}
		</Flex>
	);
}
