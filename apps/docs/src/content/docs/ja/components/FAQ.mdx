export function FAQ({ q, children }) {
	return (
		<Stack className='c--faq' g='20'>
			<Flex className='c--faq__Q' g='10' ai='c'>
				<Icon icon={Question} fz='2xl' c='purple'/>
				<Lism tag='span' fw='bold'>{q}</Lism>
			</Flex>
			<Lism className='c--faq__A' pis='10' isFlow='s'>
				{children}
			</Lism>
		</Stack>
	);
}

export function AccFAQ({ q, children }) {
	return (
		<Accordion className='c--faq' p='30' bgc='base-2' bdrs='m'>
			<AccordionHeader>
				<Flex g='10'>
					<Icon icon={Question} fz='2xl' />
					{q}
				</Flex>
			</AccordionHeader>
			<AccordionBody mbs='30' isFlow='s'>
				{children}
			</AccordionBody>
		</Accordion>
	);
}
