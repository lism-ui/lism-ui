Q&Aの構築例を紹介します。

export function Faq({ children }) {
	return (
		<Stack tag='dl' gap='20'>
			{children}
		</Stack>
	);
}

export function FaqItem({ q, children }) {
	return (
		<Stack gap='20'>
			<Flex tag='dt' gap='em6' ai='c'>
				<Icon emoji='Q' fz='xl' c='accent' />
				<Text tag='span' fw='bold'>{q}</Text>
			</Flex>
			<Lism tag='dd' px='em4' isFlow='s'>
				{children}
			</Lism>
		</Stack>
	);
}

export function FaqItemAcc({ q, children }) {
	return (
		<Accordion>
			<AccordionHeader bgc='base-3' bdrs='m'>
				<Icon icon={Question} fz='2xl' />
				<AccordionLabel>{q}</AccordionLabel>
				<AccordionIcon />
			</AccordionHeader>
			<AccordionBody>
				{children}
			</AccordionBody>
		</Accordion>
	);
}
