import {  Alert, Note, Tabs, TabItem, Spacer } from '@loos/lism-astro';
import { Reference, MemoBadge, HelpText, DammyText } from '~/components';
import { Preview, PreviewArea, PreviewTitle, PreviewCode } from '~/components/Preview';

export const NoteDemos = ({ variant }) => {
	const PRESETS = ['alert', 'point', 'warning', 'check', 'info', 'help', 'note'];
	return (
		<>
			{PRESETS.map((preset) => {
				return (
					<Note heading='Note default' variant={variant} preset={`${preset}`} key={preset}>
						<DammyText length='codes' lang='ja'/>
					</Note>
				);
			})}
		</>
	);
};

## 主なprops

- heading
- `isFlow`: `__body`に渡されます

<MemoBadge>Memo:</MemoBadge>
NoteHeader, NoteBody にわける？ → preset で色とアイコンの一括指定ができなくなる（ContextやChildren処理が必要）
## Import

```js
import { Note } from '@loos/lism-core';
```

## Usage
