SCSSの変数を上書きすることで、出力するCSSのカスタマイズが可能です。

```scss

// ブレイクポイントの変更
@use '../path/to/@loos/lism-core/scss/query' with (
	$break-points: (
		'sm': '360px',
	)
);

// prop クラスの設定変更
@use '../path/to/@loos/lism-core/scss/setting' with (
	$prop_class_data: (
		'w': (
			// importantを付ける
			important: 1,
			// @lg のユーティリティクラスも出力する
			bp: (
				lg: 1,
			),
		),


		'h': (
			// mdのユーティリティクラスを出力しない
			bp: (
				md: 0,
			),
			// bp対応クラスを出力しない
			// bp: 0,
		),
	)
);

// lism-cssの読み込み
@use '../path/to/@loos/lism-core/scss/all';

```
